# QUERIES #


# get_all_actors()
# All actors
PREFIX pred: <http://moviesDB.com/predicate/>
SELECT ?name
WHERE { 
	?movie pred:actor ?actor .
    ?actor pred:name ?name .
	}


# get_all_directors()
# All directors
PREFIX pred: <http://moviesDB.com/predicate/>

WHERE { 
	?movie pred:director ?director .
    ?director pred:name ?name .
	}


# get_movies_by_actor($a_first_name, $a_last_name)
# All movies from an actor

    ?actor pred:name "Joel Moore" .
	?movie pred:actor ?actor .
    ?movie pred:name ?mname .
	}

# searchActor($search)
# dist_searchActor($search)
# All actors with "" in their names
PREFIX pred: <http://moviesDB.com/predicate/>
SELECT ?name
WHERE { 
    ?movie pred:actor ?actor .
    ?actor pred:name ?name .
    FILTER (CONTAINS(?name, ""))
	}
